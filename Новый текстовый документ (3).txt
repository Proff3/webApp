import React from "react";

class Row extends React.Component {
    render() {
        let row = this.props.row;
        console.log(row)
        return (Object.entries(row).map(([key, value]) =>
            <input style={{ width: 'calc(50px + 20vmin)', margin: '10px', }} className={key} value={value}></input>));
    }
}

export default Row;
